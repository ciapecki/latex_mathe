\fbox{%

  {\large Faktorregel:}\\
    Einen konstanten Faktor darf man vor das Integral ziehen.\\
      \[ \int k\cdot f(x)\ dx = k\cdot \int f(x)\ dx \]
  {\large Summenregel:}\\
    Eine Summe darf man gliederweise integrieren.
    \[ \int (f(x) + g(x))\ dx = \int f(x)\ dx + \int g(x)\ dx \]
  \end{minipage}
