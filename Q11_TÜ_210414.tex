\Aufgabe{1:}
   Lösen Sie die Gleichungen:
\begin{enumerate}[label={\alph*)}]
  \item $\frac{e^{-x}}{2}-3=0$
  \item $3e^{x-1}=4$
  \item $250e^{\ln 2x}=1200$
  \item $e^{2x}-4e^x+3=0$
  \item $2e^x-3e^{-x}+5=0$
\end{enumerate}


\Aufgabe{2:}

\begin{enumerate}[label={\alph*)}]
  \item $f(x)=e^x \cdot x$
  \item $f(x)=x^3 \cdot e^x$
  \item $f(x)=\frac{e^x}{x^3}$
\end{enumerate}
